//I put this under the Step event in the player object

//create footstep counter event

var counter_footsteps=0;

//step sound event

if (keyboard_check_pressed(ord("D")) or keyboard_check_pressed(ord("A")) and (counter_footsteps==0))
{
    audio_play_sound(step_concrete_single_01, 8, true);
    counter_footsteps = 1; // number of steps to wait before trying to play the sound again
	
}

if(keyboard_check_released(ord("D")) or keyboard_check_released(ord("A")) ) 
{
	audio_stop_sound(step_concrete_single_01);
}

else if (counter_footsteps>0) counter_footsteps--;
